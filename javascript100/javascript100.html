01. 변수 

{
    var x = 100;
    var y = 200;
    var z = "javascript";

    document.write("*** 01. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

02. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가 

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    x = 300;
    y = 400;
    z += "jquery";

    document.write("
*** 02. 변수 ***
"); 
    document.write(x, "
");
    document.write(y, "
");
    document.write(z, "
");
}

03. 상수 : 데이터 저장소 

{
    const x = 100;
    const y = 200;
    const z = "javascript";
    //x = 200;

    document.write("
*** 03. 상수 ***
");
    document.write(x, "
");
    document.write(y, "
");
    document.write(z, "
");
}
